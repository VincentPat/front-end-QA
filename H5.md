### 如何修改下面的代码，使满足需求？

```
<ul id="testUL">
    <li>index = 0</li>
    <li>index = 1</li>
    <li>index = 2</li>
    <li>index = 3</li>
</ul>
<script type="text/javascript">
    var nodes = document.getElementsByTagName("li");
    for(i = 0;i < nodes.length;i += 1){
        nodes[i].onclick = function() {
            console.log(i); // 现在每次都输出4
        };
     }
 </script>
```
